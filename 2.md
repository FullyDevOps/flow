## ERD (Core Compliance Database) ‚Äî paste into `docs/erd-core.md`

```mermaid
erDiagram
  BRAND ||--o{ SKU : owns
  SKU ||--o{ SKU_PACK : "has sellable units"
  PACK_FORMAT ||--o{ SKU_PACK : "defines size/format"
  SPEC_VERSION ||--o{ SKU_PACK : "controls release criteria"
  LABEL_VERSION ||--o{ SKU_PACK : "controls label + claims"

  SUPPLIER ||--o{ RAW_LOT : provides
  SKU ||--o{ RAW_LOT : "raw material for"

  SKU_PACK ||--o{ PRODUCTION_RUN : produces
  PRODUCTION_RUN ||--|| FINISHED_LOT : creates

  FINISHED_LOT ||--o{ QC_RESULT : "tested by"
  LAB ||--o{ QC_RESULT : issues
  FINISHED_LOT ||--|| COA : "has COA"
  LAB ||--o{ COA : issues

  SHIPMENT ||--o{ SHIPMENT_LINE : contains
  FINISHED_LOT ||--o{ SHIPMENT_LINE : shipped_as
  SHIPMENT ||--o{ PALLET : uses
  PALLET ||--o{ CARTON : contains
  SKU_PACK ||--o{ CARTON : "packed as"
  FINISHED_LOT ||--o{ CARTON : "lot in cartons"

  BRAND {
    string brand_id PK
    string brand_name
    string us_entity_name
    string us_address
  }

  SKU {
    string sku_id PK
    string sku_name
    string sku_type "beetroot/orange/lemon/ginger/carrot"
    string variant "whole|peel|juice_solids"
    boolean organic_capable
    string default_mesh_range
  }

  PACK_FORMAT {
    string format_id PK
    string format_type "POUCH|JAR"
    int size_oz "8 or 16"
    int net_g
  }

  SKU_PACK {
    string sku_pack_id PK
    string sku_id FK
    string format_id FK
    string upc
    string asin
    string fnsku_strategy "label|commingle|na"
    boolean active
  }

  SPEC_VERSION {
    string spec_version_id PK
    string sku_pack_id FK
    string revision
    date effective_date
    float max_moisture_pct
    float max_aw
    string mesh_spec
    string micro_limits "TAMC/YM/Coliform/Ecoli/Salmonella"
    string metals_limits "Pb/Cd/As/Hg"
    string pesticide_policy "freq + panel"
  }

  LABEL_VERSION {
    string label_version_id PK
    string sku_pack_id FK
    string revision
    date approved_date
    string ingredient_statement
    string allergen_statement
    string net_qty_statement
    string country_of_origin_mark
    boolean nutrition_facts_included
    boolean organic_claim_on_label
  }

  SUPPLIER {
    string supplier_id PK
    string supplier_name
    string supplier_type "farm|aggregator|processor"
    string location
    boolean organic_certified
  }

  RAW_LOT {
    string raw_lot_id PK
    string sku_id FK
    string supplier_id FK
    date received_date
    float qty_kg
    boolean organic_flag
    string receiving_disposition "accept|hold|reject"
    string notes
  }

  PRODUCTION_RUN {
    string run_id PK
    string sku_pack_id FK
    date start_date
    string shift
    string dryer_id
    string mill_id
    string sifter_mesh
  }

  FINISHED_LOT {
    string finished_lot_id PK
    string run_id FK
    string lot_code
    date mfg_date
    date best_by_date
    int qty_units
    int qty_cases
    string status "quarantine|released|rejected|rework"
    string storage_location
  }

  LAB {
    string lab_id PK
    string lab_name
    string accreditation "ISO17025|other"
    string country
  }

  COA {
    string coa_id PK
    string finished_lot_id FK
    string lab_id FK
    date report_date
    string coa_doc_ref
    string qa_approver
  }

  QC_RESULT {
    string qc_result_id PK
    string finished_lot_id FK
    string lab_id FK
    string test_type "moisture|aw|mesh|TAMC|YM|Ecoli|Salmonella|metals|pesticides"
    string method_ref
    string result_value
    string unit
    string spec_limit
    boolean pass
    date tested_at
  }

  SHIPMENT {
    string shipment_id PK
    string mode "ocean|air"
    string incoterm "FOB|CIF|DDP|EXW"
    string origin_port
    string us_port
    date etd
    date eta
    string prior_notice_conf
    string isf_conf
    string broker_name
    string forwarder_name
  }

  SHIPMENT_LINE {
    string shipment_line_id PK
    string shipment_id FK
    string finished_lot_id FK
    int units
    int cases
    int pallets
  }

  PALLET {
    string pallet_id PK
    string shipment_id FK
    string pallet_label
    string pallet_type "plastic|wood"
    boolean ispm15_required
    float gross_kg
    float cubic_m
  }

  CARTON {
    string carton_id PK
    string pallet_id FK
    string sku_pack_id FK
    string finished_lot_id FK
    string carton_label
    int units_per_carton
    string lot_code_printed
    date best_by_printed
  }
```
